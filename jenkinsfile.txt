pipeline {
  agent any
  stages {
    stage('Clone Repo') {
      steps {
        git 'https://github.com/your-username/aws-devops-ci-cd-project.git'
      }
    }
    stage('Build Docker Image') {
      steps {
        sh 'docker build -t my-node-app ./docker'
      }
    }
    stage('Run Docker Container') {
      steps {
        sh 'docker run -d -p 3000:3000 my-node-app'
      }
    }
  }
}
